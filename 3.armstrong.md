import java.util.*;

class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int num = sc.nextInt();
        int original = num;
        int sumCube = 0;

        while(num != 0){
            int lastnum = num % 10;
            sumCube = sumCube + (lastnum * lastnum * lastnum);
            num = num / 10;
        }

        if(original == sumCube){
            System.out.println("True");
        }else{
            System.out.println("True");
        }

        sc.close();
    }
}
